99 // Copied from strided_range thrust example by Nathan Bell.
100 // Iterator to move forward by a specified number of steps
101 // in each iteration.
102 template<typename Iterator>
103 class strided_range {
104  public:
105  typedef typename thrust::iterator_difference<Iterator>::type difference_type;
106 
107  struct stride_functor : public thrust::unary_function<difference_type, difference_type> {
108  difference_type stride;
109 
110  stride_functor(difference_type stride)
111  : stride(stride) {}
112 
113  __host__ __device__ difference_type operator()(const difference_type &i) const { return stride * i; }
114  };
115  typedef typename thrust::counting_iterator<difference_type> CountingIterator;
116  typedef typename thrust::transform_iterator<stride_functor, CountingIterator> TransformIterator;
117  typedef typename thrust::permutation_iterator<Iterator, TransformIterator> PermutationIterator;
118 
119  // type of the strided_range iterator
120  typedef PermutationIterator iterator;
121 
122  // construct strided_range for the range [first,last)
123  strided_range(Iterator first, Iterator last, difference_type stride)
124  : first(first)
125  , last(last)
126  , stride(stride) {}
127 
128  iterator begin() const {
129  return PermutationIterator(first, TransformIterator(CountingIterator(0), stride_functor(stride)));
130  }
131 
132  iterator end() const { return begin() + ((last - first) + (stride - 1)) / stride; }
133 
134  protected:
135  Iterator first;
136  Iterator last;
137  difference_type stride;
138 };
