69 // Copied from strided_range thrust example by Nathan Bell.
70 // Iterator to move forward by a specified number of steps
71 // in each iteration.
72 template <typename Iterator> class strided_range {
73 public:
74  typedef typename thrust::iterator_difference<Iterator>::type difference_type;
75 
76  struct stride_functor : public thrust::unary_function<difference_type, difference_type> {
77  difference_type stride;
78 
79  stride_functor(difference_type stride)
80  : stride(stride) {}
81 
82  __host__ __device__ difference_type operator()(const difference_type& i) const {
83  return stride * i;
84  }
85  };
86  typedef typename thrust::counting_iterator<difference_type> CountingIterator;
87  typedef typename thrust::transform_iterator<stride_functor, CountingIterator> TransformIterator;
88  typedef typename thrust::permutation_iterator<Iterator, TransformIterator> PermutationIterator;
89 
90  // type of the strided_range iterator
91  typedef PermutationIterator iterator;
92 
93  // construct strided_range for the range [first,last)
94  strided_range(Iterator first, Iterator last, difference_type stride)
95  : first(first), last(last), stride(stride) {}
96 
97  iterator begin() const {
98  return PermutationIterator(first, TransformIterator(CountingIterator(0), stride_functor(stride)));
99  }
100 
101  iterator end() const {
102  return begin() + ((last - first) + (stride - 1)) / stride;
103  }
104 
105 protected:
106  Iterator first;
107  Iterator last;
108  difference_type stride;
109 };
