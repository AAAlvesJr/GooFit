66 double TRandom3::Rndm(int) {
67 // Machine independent random number generator.
68 // Produces uniformly-distributed floating points in ]0,1]
69 // Method: Mersenne Twistor
70 
71  unsigned int y;
72 
73  const int kM = 397;
74  const int kN = 624;
75  const unsigned int kTemperingMaskB = 0x9d2c5680;
76  const unsigned int kTemperingMaskC = 0xefc60000;
77  const unsigned int kUpperMask = 0x80000000;
78  const unsigned int kLowerMask = 0x7fffffff;
79  const unsigned int kMatrixA = 0x9908b0df;
80 
81  if(fCount624 >= kN) {
82  register int i;
83 
84  for(i=0; i < kN-kM; i++) {
85  y = (fMt[i] & kUpperMask) | (fMt[i+1] & kLowerMask);
86  fMt[i] = fMt[i+kM] ^ (y >> 1) ^ ((y & 0x1) ? kMatrixA : 0x0);
87  }
88 
89  for(; i < kN-1 ; i++) {
90  y = (fMt[i] & kUpperMask) | (fMt[i+1] & kLowerMask);
91  fMt[i] = fMt[i+kM-kN] ^ (y >> 1) ^ ((y & 0x1) ? kMatrixA : 0x0);
92  }
93 
94  y = (fMt[kN-1] & kUpperMask) | (fMt[0] & kLowerMask);
95  fMt[kN-1] = fMt[kM-1] ^ (y >> 1) ^ ((y & 0x1) ? kMatrixA : 0x0);
96  fCount624 = 0;
97  }
98 
99  y = fMt[fCount624++];
100  y ^= (y >> 11);
101  y ^= ((y << 7) & kTemperingMaskB);
102  y ^= ((y << 15) & kTemperingMaskC);
103  y ^= (y >> 18);
104 
105  if(y)
106  return ((double) y * 2.3283064365386963e-10); // * Power(2,-32)
107 
108  return Rndm();
109 }

120 void TRandom3::RndmArray(int n, double* array) {
121  // Return an array of n random numbers uniformly distributed in ]0,1]
122 
123  int k = 0;
124 
125  unsigned int y;
126 
127  const int kM = 397;
128  const int kN = 624;
129  const unsigned int kTemperingMaskB = 0x9d2c5680;
130  const unsigned int kTemperingMaskC = 0xefc60000;
131  const unsigned int kUpperMask = 0x80000000;
132  const unsigned int kLowerMask = 0x7fffffff;
133  const unsigned int kMatrixA = 0x9908b0df;
134 
135  while(k < n) {
136  if(fCount624 >= kN) {
137  register int i;
138 
139  for(i=0; i < kN-kM; i++) {
140  y = (fMt[i] & kUpperMask) | (fMt[i+1] & kLowerMask);
141  fMt[i] = fMt[i+kM] ^ (y >> 1) ^ ((y & 0x1) ? kMatrixA : 0x0);
142  }
143 
144  for(; i < kN-1 ; i++) {
145  y = (fMt[i] & kUpperMask) | (fMt[i+1] & kLowerMask);
146  fMt[i] = fMt[i+kM-kN] ^ (y >> 1) ^ ((y & 0x1) ? kMatrixA : 0x0);
147  }
148 
149  y = (fMt[kN-1] & kUpperMask) | (fMt[0] & kLowerMask);
150  fMt[kN-1] = fMt[kM-1] ^ (y >> 1) ^ ((y & 0x1) ? kMatrixA : 0x0);
151  fCount624 = 0;
152  }
153 
154  y = fMt[fCount624++];
155  y ^= (y >> 11);
156  y ^= ((y << 7) & kTemperingMaskB);
157  y ^= ((y << 15) & kTemperingMaskC);
158  y ^= (y >> 18);
159 
160  if(y) {
161  array[k] = double(y * 2.3283064365386963e-10); // * Power(2,-32)
162  k++;
163  }
164  }
165 }
