1 __device__ fptype twoBodyCMmom (double rMassSq, fptype d1m, fptype d2m) {
2  // For A -> B + C, calculate momentum of B and C in rest frame of A.
3  // PDG 38.16.
4 
5  fptype kin1 = 1 - POW(d1m+d2m, 2) / rMassSq;

25 __device__ bool inDalitz (fptype m12, fptype m13, fptype bigM, fptype dm1, fptype dm2, fptype dm3) {
26  if (m12 < POW(dm1 + dm2, 2)) return false; // This m12 cannot exist, it's less than the square of the (1,2) particle mass.
27  if (m12 > POW(bigM - dm3, 2)) return false; // This doesn't work either, there's no room for an at-rest 3 daughter.
28 
29  // Calculate energies of 1 and 3 particles in m12 rest frame.
30  fptype e1star = 0.5 * (m12 - dm2*dm2 + dm1*dm1) / SQRT(m12);
31  fptype e3star = 0.5 * (bigM*bigM - m12 - dm3*dm3) / SQRT(m12);
32 
33  // Bounds for m13 at this value of m12.
34  fptype minimum = POW(e1star + e3star, 2) - POW(SQRT(e1star*e1star - dm1*dm1) + SQRT(e3star*e3star - dm3*dm3), 2);
35  if (m13 < minimum) return false;
36  fptype maximum = POW(e1star + e3star, 2) - POW(SQRT(e1star*e1star - dm1*dm1) - SQRT(e3star*e3star - dm3*dm3), 2);
37  if (m13 > maximum) return false;
38 
39  return true;
40 }
