565 __device__ fptype MetricTaker::operator () (thrust::tuple<int, fptype*, int> t) const {
566  // Calculate event offset for this thread.
567  int eventIndex = thrust::get<0>(t);
568  int eventSize = thrust::get<2>(t);
569  fptype* eventAddress = thrust::get<1>(t) + (eventIndex * abs(eventSize));
570 
571  // Causes stack size to be statically undeterminable.
572  fptype ret = callFunction(eventAddress, functionIdx, parameters);
573  //fptype ret = (*(reinterpret_cast<device_function_ptr>(device_function_table[functionIdx])))(eventAddress, cudaArray, paramIndices + parameters);

575  // Notice assumption here! For unbinned fits the 'eventAddress' pointer won't be used
576  // in the metric, so it doesn't matter what it is. For binned fits it is assumed that
577  // the structure of the event is (obs1 obs2... binentry binvolume), so that the array
578  // passed to the metric consists of (binentry binvolume).
579  ret = (*(reinterpret_cast<device_metric_ptr>(device_function_table[metricIndex])))(ret, eventAddress + (abs(eventSize)-2), parameters);

585 #define MAX_NUM_OBSERVABLES 5
586 __device__ fptype MetricTaker::operator () (thrust::tuple<int, int, fptype*> t) const {
587  // Bin index, event size, base address [lower, upper, numbins]
588 
589  int evtSize = thrust::get<1>(t);
590  int binNumber = thrust::get<0>(t);
591 
592  // Do not understand why this cannot be declared __shared__. Dynamically allocating shared memory is apparently complicated.
593  //fptype* binCenters = (fptype*) malloc(evtSize * sizeof(fptype));
594  __shared__ fptype binCenters[1024*MAX_NUM_OBSERVABLES];
595 
596  // To convert global bin number to (x,y,z...) coordinates: For each dimension, take the mod
597  // with the number of bins in that dimension. Then divide by the number of bins, in effect
598  // collapsing so the grid has one fewer dimension. Rinse and repeat.
599  unsigned int* indices = paramIndices + parameters;
600  for (int i = 0; i < evtSize; ++i) {
601  fptype lowerBound = thrust::get<2>(t)[3*i+0];
602  fptype upperBound = thrust::get<2>(t)[3*i+1];
603  int numBins = (int) FLOOR(thrust::get<2>(t)[3*i+2] + 0.5);
604  int localBin = binNumber % numBins;
605 
606  fptype x = upperBound - lowerBound;
607  x /= numBins;
608  x *= (localBin + 0.5);
609  x += lowerBound;
610  binCenters[indices[indices[0] + 2 + i]+threadIdx.x*MAX_NUM_OBSERVABLES] = x;
611  binNumber /= numBins;
612 
613  //if (gpuDebug & 1)
614  //if ((gpuDebug & 1) && (0 == threadIdx.x) && (0 == blockIdx.x))
615  //printf("[%i, %i] Bins: %i %i %i %f %f %f %f %i\n", blockIdx.x, threadIdx.x, binNumber, numBins, localBin, x, lowerBound, upperBound, thrust::get<2>(t)[3*i+2], indices[indices[0] + 2 + i]);
616  //printf("Bins: %i %i %i %f %f\n", i, indices[indices[0] + 2 + i]+threadIdx.x*MAX_NUM_OBSERVABLES, indices[indices[0] + 2 + i], x, binCenters[threadIdx.x*MAX_NUM_OBSERVABLES]);
617  }
618 
619  // Causes stack size to be statically undeterminable.
620  //fptype ret = (*(reinterpret_cast<device_function_ptr>(device_function_table[functionIdx])))(binCenters+threadIdx.x*MAX_NUM_OBSERVABLES, cudaArray, indices);
621  fptype ret = callFunction(binCenters+threadIdx.x*MAX_NUM_OBSERVABLES, functionIdx, parameters);
622  //if (gpuDebug & 1) printf("[%i, %i] Binned eval: %f %f\n", blockIdx.x, threadIdx.x, binCenters[threadIdx.x*4], ret);
623  return ret;
624 }
