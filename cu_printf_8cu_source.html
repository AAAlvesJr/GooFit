12 /*
13  * cuPrintf.cu
14  *
15  * This is a printf command callable from within a kernel. It is set
16  * up so that output is sent to a memory buffer, which is emptied from
17  * the host side - but only after a cudaThreadSynchronize() on the host.
18  *
19  * Currently, there is a limitation of around 200 characters of output
20  * and no more than 10 arguments to a single cuPrintf() call. Issue
21  * multiple calls if longer format strings are required.
22  *
23  * It requires minimal setup, and is *NOT* optimised for performance.
24  * For example, writes are not coalesced - this is because there is an
25  * assumption that people will not want to printf from every single one
26  * of thousands of threads, but only from individual threads at a time.
27  *
28  * Using this is simple - it requires one host-side call to initialise
29  * everything, and then kernels can call cuPrintf at will. Sample code
30  * is the easiest way to demonstrate:
31  *
32  #include "cuPrintf.cu"
33 
34  __global__ void testKernel(int val)
35  {
36  cuPrintf("Value is: %d\n", val);
37  }
38 
39  int main()
40  {
41  cudaPrintfInit();
42  testKernel<<< 2, 3 >>>(10);
43  cudaPrintfDisplay(stdout, true);
44  cudaPrintfEnd();
45  return 0;
46  }
47  *
48  * See the header file, "cuPrintf.cuh" for more info, especially
49  * arguments to cudaPrintfInit() and cudaPrintfDisplay();
50  */

70 // The main storage is in a global print buffer, which has a known
71 // start/end/length. These are atomically updated so it works as a
72 // circular buffer.
73 // Since the only control primitive that can be used is atomicAdd(),
74 // we cannot wrap the pointer as such. The actual address must be
75 // calculated from printfBufferPtr by mod-ing with printfBufferLength.
76 // For sm_10 architecture, we must subdivide the buffer per-thread
77 // since we do not even have an atomic primitive.
78 __constant__ static char *globalPrintfBuffer = NULL; // Start of circular buffer (set up by host)
79 __constant__ static int printfBufferLength = 0; // Size of circular buffer (set up by host)
80 __device__ static cuPrintfRestriction restrictRules; // Output restrictions
81 __device__ volatile static char *printfBufferPtr = NULL; // Current atomically-incremented non-wrapped offset

113 //
114 // getNextPrintfBufPtr
115 //
116 // Grabs a block of space in the general circular buffer, using an
117 // atomic function to ensure that it's ours. We handle wrapping
118 // around the circular buffer and return a pointer to a place which
119 // can be written to.
120 //
121 // Important notes:
122 // 1. We always grab CUPRINTF_MAX_LEN bytes
123 // 2. Because of 1, we never worry about wrapping around the end
124 // 3. Because of 1, printfBufferLength *must* be a factor of CUPRINTF_MAX_LEN
125 //
126 // This returns a pointer to the place where we own.
127 //
128 __device__ static char *getNextPrintfBufPtr()
129 {
130  // Initialisation check
131  if(!printfBufferPtr)
132  return NULL;
133 
134  // Thread/block restriction check
135  if((restrictRules.blockid != CUPRINTF_UNRESTRICTED) && (restrictRules.blockid != (blockIdx.x + gridDim.x*blockIdx.y)))
136  return NULL;
137  if((restrictRules.threadid != CUPRINTF_UNRESTRICTED) && (restrictRules.threadid != (threadIdx.x + blockDim.x*threadIdx.y + blockDim.x*blockDim.y*threadIdx.z)))
138  return NULL;
139 
140  // Conditional section, dependent on architecture
141 #if __CUDA_ARCH__ == 100
142  // For sm_10 architectures, we have no atomic add - this means we must split the
143  // entire available buffer into per-thread blocks. Inefficient, but what can you do.
144  int thread_count = (gridDim.x * gridDim.y) * (blockDim.x * blockDim.y * blockDim.z);
145  int thread_index = threadIdx.x + blockDim.x*threadIdx.y + blockDim.x*blockDim.y*threadIdx.z +
146  (blockIdx.x + gridDim.x*blockIdx.y) * (blockDim.x * blockDim.y * blockDim.z);
147 
148  // Find our own block of data and go to it. Make sure the per-thread length
149  // is a precise multiple of CUPRINTF_MAX_LEN, otherwise we risk size and
150  // alignment issues! We must round down, of course.
151  unsigned int thread_buf_len = printfBufferLength / thread_count;
152  thread_buf_len &= ~(CUPRINTF_MAX_LEN-1);
153 
154  // We *must* have a thread buffer length able to fit at least two printfs (one header, one real)
155  if(thread_buf_len < (CUPRINTF_MAX_LEN * 2))
156  return NULL;
157 
158  // Now address our section of the buffer. The first item is a header.
159  char *myPrintfBuffer = globalPrintfBuffer + (thread_buf_len * thread_index);
160  cuPrintfHeaderSM10 hdr = *(cuPrintfHeaderSM10 *)(void *)myPrintfBuffer;
161  if(hdr.magic != CUPRINTF_SM10_MAGIC)
162  {
163  // If our header is not set up, initialise it
164  hdr.magic = CUPRINTF_SM10_MAGIC;
165  hdr.thread_index = thread_index;
166  hdr.thread_buf_len = thread_buf_len;
167  hdr.offset = 0; // Note we start at 0! We pre-increment below.
168  *(cuPrintfHeaderSM10 *)(void *)myPrintfBuffer = hdr; // Write back the header
169 
170  // For initial setup purposes, we might need to init thread0's header too
171  // (so that cudaPrintfDisplay() below will work). This is only run once.
172  cuPrintfHeaderSM10 *tophdr = (cuPrintfHeaderSM10 *)(void *)globalPrintfBuffer;
173  tophdr->thread_buf_len = thread_buf_len;
174  }
175 
176  // Adjust the offset by the right amount, and wrap it if need be
177  unsigned int offset = hdr.offset + CUPRINTF_MAX_LEN;
178  if(offset >= hdr.thread_buf_len)
179  offset = CUPRINTF_MAX_LEN;
180 
181  // Write back the new offset for next time and return a pointer to it
182  ((cuPrintfHeaderSM10 *)(void *)myPrintfBuffer)->offset = offset;
183  return myPrintfBuffer + offset;
184 #else
185  // Much easier with an atomic operation!
186  size_t offset = atomicAdd((unsigned int *)&printfBufferPtr, CUPRINTF_MAX_LEN) - (size_t)globalPrintfBuffer;
187  offset %= printfBufferLength;
188  return globalPrintfBuffer + offset;
189 #endif
190 }

214 //
215 // cuPrintfStrncpy
216 //
217 // This special strncpy outputs an aligned length value, followed by the
218 // string. It then zero-pads the rest of the string until a 64-aligned
219 // boundary. The length *includes* the padding. A pointer to the byte
220 // just after the \0 is returned.
221 //
222 // This function could overflow CUPRINTF_MAX_LEN characters in our buffer.
223 // To avoid it, we must count as we output and truncate where necessary.
224 //
225 __device__ static char *cuPrintfStrncpy(char *dest, const char *src, int n, char *end)
226 {
227  // Initialisation and overflow check
228  if(!dest || !src || (dest >= end))
229  return NULL;
230 
231  // Prepare to write the length specifier. We're guaranteed to have
232  // at least "CUPRINTF_ALIGN_SIZE" bytes left because we only write out in
233  // chunks that size, and CUPRINTF_MAX_LEN is aligned with CUPRINTF_ALIGN_SIZE.
234  int *lenptr = (int *)(void *)dest;
235  int len = 0;
236  dest += CUPRINTF_ALIGN_SIZE;
237 
238  // Now copy the string
239  while(n--)
240  {
241  if(dest >= end) // Overflow check
242  break;
243 
244  len++;
245  *dest++ = *src;
246  if(*src++ == '\0')
247  break;
248  }
249 
250  // Now write out the padding bytes, and we have our length.
251  while((dest < end) && (((size_t)dest & (CUPRINTF_ALIGN_SIZE-1)) != 0))
252  {
253  len++;
254  *dest++ = 0;
255  }
256  *lenptr = len;
257  return (dest < end) ? dest : NULL; // Overflow means return NULL
258 }

262 // copyArg
263 //
264 // This copies a length specifier and then the argument out to the
265 // data buffer. Templates let the compiler figure all this out at
266 // compile-time, making life much simpler from the programming
267 // point of view. I'm assuimg all (const char *) is a string, and
268 // everything else is the variable it points at. I'd love to see
269 // a better way of doing it, but aside from parsing the format
270 // string I can't think of one.
271 //
272 // The length of the data type is inserted at the beginning (so that
273 // the display can distinguish between float and double), and the
274 // pointer to the end of the entry is returned.
275 //
276 __device__ static char *copyArg(char *ptr, const char *arg, char *end)
277 {
278  // Initialisation check
279  if(!ptr || !arg)
280  return NULL;
281 
282  // strncpy does all our work. We just terminate.
283  if((ptr = cuPrintfStrncpy(ptr, arg, CUPRINTF_MAX_LEN, end)) != NULL)
284  *ptr = 0;
285 
286  return ptr;
287 }

289 template <typename T>
290 __device__ static char *copyArg(char *ptr, T &arg, char *end)
291 {
292  // Initisalisation and overflow check. Alignment rules mean that
293  // we're at least CUPRINTF_ALIGN_SIZE away from "end", so we only need
294  // to check that one offset.
295  if(!ptr || ((ptr+CUPRINTF_ALIGN_SIZE) >= end))
296  return NULL;
297 
298  // Write the length and argument
299  *(int *)(void *)ptr = sizeof(arg);
300  ptr += CUPRINTF_ALIGN_SIZE;
301  *(T *)(void *)ptr = arg;
302  ptr += CUPRINTF_ALIGN_SIZE;
303  *ptr = 0;
304 
305  return ptr;
306 }

309 //
310 // cuPrintf
311 //
312 // Templated printf functions to handle multiple arguments.
313 // Note we return the total amount of data copied, not the number
314 // of characters output. But then again, who ever looks at the
315 // return from printf() anyway?
316 //
317 // The format is to grab a block of circular buffer space, the
318 // start of which will hold a header and a pointer to the format
319 // string. We then write in all the arguments, and finally the
320 // format string itself. This is to make it easy to prevent
321 // overflow of our buffer (we support up to 10 arguments, each of
322 // which can be 12 bytes in length - that means that only the
323 // format string (or a %s) can actually overflow; so the overflow
324 // check need only be in the strcpy function.
325 //
326 // The header is written at the very last because that's what
327 // makes it look like we're done.
328 //
329 // Errors, which are basically lack-of-initialisation, are ignored
330 // in the called functions because NULL pointers are passed around
331 //
332 
333 // All printf variants basically do the same thing, setting up the
334 // buffer, writing all arguments, then finalising the header. For
335 // clarity, we'll pack the code into some big macros.
336 #define CUPRINTF_PREAMBLE \
337  char *start, *end, *bufptr, *fmtstart; \
338  if((start = getNextPrintfBufPtr()) == NULL) return 0; \
339  end = start + CUPRINTF_MAX_LEN; \
340  bufptr = start + sizeof(cuPrintfHeader);
341 
342 // Posting an argument is easy
343 #define CUPRINTF_ARG(argname) \
344  bufptr = copyArg(bufptr, argname, end);
345 
346 // After args are done, record start-of-fmt and write the fmt and header
347 #define CUPRINTF_POSTAMBLE \
348  fmtstart = bufptr; \
349  end = cuPrintfStrncpy(bufptr, fmt, CUPRINTF_MAX_LEN, end); \
350  writePrintfHeader(start, end ? fmtstart : NULL); \
351  return end ? (int)(end - start) : 0;

521 //
522 // outputPrintfData
523 //
524 // Our own internal function, which takes a pointer to a data buffer
525 // and passes it through libc's printf for output.
526 //
527 // We receive the formate string and a pointer to where the data is
528 // held. We then run through and print it out.
529 //
530 // Returns 0 on failure, 1 on success
531 //
532 static int outputPrintfData(char *fmt, char *data)
533 {
534  // Format string is prefixed by a length that we don't need
535  fmt += CUPRINTF_ALIGN_SIZE;
536 
537  // Now run through it, printing everything we can. We must
538  // run to every % character, extract only that, and use printf
539  // to format it.
540  char *p = strchr(fmt, '%');
541  while(p != NULL)
542  {
543  // Print up to the % character
544  *p = '\0';
545  fputs(fmt, printf_fp);
546  *p = '%'; // Put back the %
547 
548  // Now handle the format specifier
549  char *format = p++; // Points to the '%'
550  p += strcspn(p, "%cdiouxXeEfgGaAnps");
551  if(*p == '\0') // If no format specifier, print the whole thing
552  {
553  fmt = format;
554  break;
555  }
556 
557  // Cut out the format bit and use printf to print it. It's prefixed
558  // by its length.
559  int arglen = *(int *)data;
560  if(arglen > CUPRINTF_MAX_LEN)
561  {
562  fputs("Corrupt printf buffer data - aborting\n", printf_fp);
563  return 0;
564  }
565 
566  data += CUPRINTF_ALIGN_SIZE;
567 
568  char specifier = *p++;
569  char c = *p; // Store for later
570  *p = '\0';
571  switch(specifier)
572  {
573  // These all take integer arguments
574  case 'c':
575  case 'd':
576  case 'i':
577  case 'o':
578  case 'u':
579  case 'x':
580  case 'X':
581  case 'p':
582  fprintf(printf_fp, format, *((int *)data));
583  break;
584 
585  // These all take double arguments
586  case 'e':
587  case 'E':
588  case 'f':
589  case 'g':
590  case 'G':
591  case 'a':
592  case 'A':
593  if(arglen == 4) // Float vs. Double thing
594  fprintf(printf_fp, format, *((float *)data));
595  else
596  fprintf(printf_fp, format, *((double *)data));
597  break;
598 
599  // Strings are handled in a special way
600  case 's':
601  fprintf(printf_fp, format, (char *)data);
602  break;
603 
604  // % is special
605  case '%':
606  fprintf(printf_fp, "%%");
607  break;
608 
609  // Everything else is just printed out as-is
610  default:
611  fprintf(printf_fp, "%s", format);
612  break;
613  }
614  data += CUPRINTF_ALIGN_SIZE; // Move on to next argument
615  *p = c; // Restore what we removed
616  fmt = p; // Adjust fmt string to be past the specifier
617  p = strchr(fmt, '%'); // and get the next specifier
618  }
619 
620  // Print out the last of the string
621  fputs(fmt, printf_fp);
622  return 1;

626 //
627 // doPrintfDisplay
628 //
629 // This runs through the blocks of CUPRINTF_MAX_LEN-sized data, calling the
630 // print function above to display them. We've got this separate from
631 // cudaPrintfDisplay() below so we can handle the SM_10 architecture
632 // partitioning.
633 //
634 static int doPrintfDisplay(int headings, int clear, char *bufstart, char *bufend, char *bufptr, char *endptr)
635 {
636  // Grab, piece-by-piece, each output element until we catch
637  // up with the circular buffer end pointer
638  int printf_count=0;
639  char printfbuf_local[CUPRINTF_MAX_LEN+1];
640  printfbuf_local[CUPRINTF_MAX_LEN] = '\0';
641 
642  while(bufptr != endptr)
643  {
644  // Wrap ourselves at the end-of-buffer
645  if(bufptr == bufend)
646  bufptr = bufstart;
647 
648  // Adjust our start pointer to within the circular buffer and copy a block.
649  cudaMemcpy(printfbuf_local, bufptr, CUPRINTF_MAX_LEN, cudaMemcpyDeviceToHost);
650 
651  // If the magic number isn't valid, then this write hasn't gone through
652  // yet and we'll wait until it does (or we're past the end for non-async printfs).
653  cuPrintfHeader *hdr = (cuPrintfHeader *)printfbuf_local;
654  if((hdr->magic != CUPRINTF_SM11_MAGIC) || (hdr->fmtoffset >= CUPRINTF_MAX_LEN))
655  {
656  //fprintf(printf_fp, "Bad magic number in printf header\n");
657  break;
658  }
659 
660  // Extract all the info and get this printf done
661  if(headings)
662  fprintf(printf_fp, "[%d, %d]: ", hdr->blockid, hdr->threadid);
663  if(hdr->fmtoffset == 0)
664  fprintf(printf_fp, "printf buffer overflow\n");
665  else if(!outputPrintfData(printfbuf_local+hdr->fmtoffset, printfbuf_local+sizeof(cuPrintfHeader)))
666  break;
667  printf_count++;
668 
669  // Clear if asked
670  if(clear)
671  cudaMemset(bufptr, 0, CUPRINTF_MAX_LEN);
672 
673  // Now advance our start location, because we're done, and keep copying
674  bufptr += CUPRINTF_MAX_LEN;
675  }
676 
677  return printf_count;
678 }

681 //
682 // cudaPrintfInit
683 //
684 // Takes a buffer length to allocate, creates the memory on the device and
685 // returns a pointer to it for when a kernel is called. It's up to the caller
686 // to free it.
687 //
688 extern "C" cudaError_t cudaPrintfInit(size_t bufferLen)
689 {
690  // Fix up bufferlen to be a multiple of CUPRINTF_MAX_LEN
691  bufferLen = (bufferLen < (size_t)CUPRINTF_MAX_LEN) ? CUPRINTF_MAX_LEN : bufferLen;
692  if((bufferLen % CUPRINTF_MAX_LEN) > 0)
693  bufferLen += (CUPRINTF_MAX_LEN - (bufferLen % CUPRINTF_MAX_LEN));
694  printfbuf_len = (int)bufferLen;
695 
696  // Allocate a print buffer on the device and zero it
697  if(cudaMalloc((void **)&printfbuf_device, printfbuf_len) != cudaSuccess)
698  return cudaErrorInitializationError;
699  cudaMemset(printfbuf_device, 0, printfbuf_len);
700  printfbuf_start = printfbuf_device; // Where we start reading from
701 
702  // No restrictions to begin with
703  cuPrintfRestriction restrict;
704  restrict.threadid = restrict.blockid = CUPRINTF_UNRESTRICTED;
705  cudaMemcpyToSymbol(restrictRules, &restrict, sizeof(restrict));
706 
707  // Initialise the buffer and the respective lengths/pointers.
708  cudaMemcpyToSymbol(globalPrintfBuffer, &printfbuf_device, sizeof(char *));
709  cudaMemcpyToSymbol(printfBufferPtr, &printfbuf_device, sizeof(char *));
710  cudaMemcpyToSymbol(printfBufferLength, &printfbuf_len, sizeof(printfbuf_len));
711 
712  return cudaSuccess;
713 }

731 //
732 // cudaPrintfDisplay
733 //
734 // Each call to this function dumps the entire current contents
735 // of the printf buffer to the pre-specified FILE pointer. The
736 // circular "start" pointer is advanced so that subsequent calls
737 // dumps only new stuff.
738 //
739 // In the case of async memory access (via streams), call this
740 // repeatedly to keep trying to empty the buffer. If it's a sync
741 // access, then the whole buffer should empty in one go.
742 //
743 // Arguments:
744 // outputFP - File descriptor to output to (NULL => stdout)
745 // showThreadID - If true, prints [block,thread] before each line
746 //
747 extern "C" cudaError_t cudaPrintfDisplay(void *outputFP, bool showThreadID)
748 {
749  printf_fp = (FILE *)((outputFP == NULL) ? stdout : outputFP);
750 
751  // For now, we force "synchronous" mode which means we're not concurrent
752  // with kernel execution. This also means we don't need clearOnPrint.
753  // If you're patching it for async operation, here's where you want it.
754  bool sync_printfs = true;
755  bool clearOnPrint = false;
756 
757  // Initialisation check
758  if(!printfbuf_start || !printfbuf_device || !printf_fp)
759  return cudaErrorMissingConfiguration;
760 
761  // To determine which architecture we're using, we read the
762  // first short from the buffer - it'll be the magic number
763  // relating to the version.
764  unsigned short magic;
765  cudaMemcpy(&magic, printfbuf_device, sizeof(unsigned short), cudaMemcpyDeviceToHost);
766 
767  // For SM_10 architecture, we've split our buffer into one-per-thread.
768  // That means we must do each thread block separately. It'll require
769  // extra reading. We also, for now, don't support async printfs because
770  // that requires tracking one start pointer per thread.
771  if(magic == CUPRINTF_SM10_MAGIC)
772  {
773  sync_printfs = true;
774  clearOnPrint = false;
775  int blocklen = 0;
776  char *blockptr = printfbuf_device;
777  while(blockptr < (printfbuf_device + printfbuf_len))
778  {
779  cuPrintfHeaderSM10 hdr;
780  cudaMemcpy(&hdr, blockptr, sizeof(hdr), cudaMemcpyDeviceToHost);
781 
782  // We get our block-size-step from the very first header
783  if(hdr.thread_buf_len != 0)
784  blocklen = hdr.thread_buf_len;
785 
786  // No magic number means no printfs from this thread
787  if(hdr.magic != CUPRINTF_SM10_MAGIC)
788  {
789  if(blocklen == 0)
790  {
791  fprintf(printf_fp, "No printf headers found at all!\n");
792  break; // No valid headers!
793  }
794  blockptr += blocklen;
795  continue;
796  }
797 
798  // "offset" is non-zero then we can print the block contents
799  if(hdr.offset > 0)
800  {
801  // For synchronous printfs, we must print from endptr->bufend, then from start->end
802  if(sync_printfs)
803  doPrintfDisplay(showThreadID, clearOnPrint, blockptr+CUPRINTF_MAX_LEN, blockptr+hdr.thread_buf_len, blockptr+hdr.offset+CUPRINTF_MAX_LEN, blockptr+hdr.thread_buf_len);
804  doPrintfDisplay(showThreadID, clearOnPrint, blockptr+CUPRINTF_MAX_LEN, blockptr+hdr.thread_buf_len, blockptr+CUPRINTF_MAX_LEN, blockptr+hdr.offset+CUPRINTF_MAX_LEN);
805  }
806 
807  // Move on to the next block and loop again
808  blockptr += hdr.thread_buf_len;
809  }
810  }
811  // For SM_11 and up, everything is a single buffer and it's simple
812  else if(magic == CUPRINTF_SM11_MAGIC)
813  {
814  // Grab the current "end of circular buffer" pointer.
815  char *printfbuf_end = NULL;
816  cudaMemcpyFromSymbol(&printfbuf_end, printfBufferPtr, sizeof(char *));
817 
818  // Adjust our starting and ending pointers to within the block
819  char *bufptr = ((printfbuf_start - printfbuf_device) % printfbuf_len) + printfbuf_device;
820  char *endptr = ((printfbuf_end - printfbuf_device) % printfbuf_len) + printfbuf_device;
821 
822  // For synchronous (i.e. after-kernel-exit) printf display, we have to handle circular
823  // buffer wrap carefully because we could miss those past "end".
824  if(sync_printfs)
825  doPrintfDisplay(showThreadID, clearOnPrint, printfbuf_device, printfbuf_device+printfbuf_len, endptr, printfbuf_device+printfbuf_len);
826  doPrintfDisplay(showThreadID, clearOnPrint, printfbuf_device, printfbuf_device+printfbuf_len, bufptr, endptr);
827 
828  printfbuf_start = printfbuf_end;
829  }
830  else
831  ;//printf("Bad magic number in cuPrintf buffer header\n");
832 
833  // If we were synchronous, then we must ensure that the memory is cleared on exit
834  // otherwise another kernel launch with a different grid size could conflict.
835  if(sync_printfs)
836  cudaMemset(printfbuf_device, 0, printfbuf_len);
837 
838  return cudaSuccess;
839 }
